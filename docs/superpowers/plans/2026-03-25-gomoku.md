# Gomoku Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a small desktop-only browser Gomoku game for Chrome on macOS with a 19x19 board, local two-player play, undo, restart, current-player status, and win detection.

**Architecture:** Use a zero-build static site with one HTML entry point, one stylesheet, and a small set of ES modules. Keep game rules and state transitions in a pure module so they can be tested in the browser with a tiny native test runner, while canvas rendering and UI wiring stay in separate focused modules.

**Tech Stack:** HTML, CSS, JavaScript (ES modules), Canvas API, browser-native test runner page

---

## File Structure

- `index.html`
  Loads the layout shell, canvas, controls, and ES module entry point.
- `style.css`
  Defines the desktop page layout, board styling, buttons, and status panel.
- `src/game.js`
  Owns pure game-state creation, move placement, undo, restart, and winner detection.
- `src/render.js`
  Owns all canvas drawing: board background, grid, star points, stones, and last-move marker.
- `src/app.js`
  Connects DOM events to state transitions and triggers redraws and status updates.
- `tests/test-runner.html`
  Runs browser-native tests without Node or a build step.
- `tests/test-framework.js`
  Provides tiny assertion helpers and DOM reporting for tests.
- `tests/game.test.js`
  Verifies pure game logic, move validation, undo, restart, and win detection.
- `README.md`
  Explains how to open the game and the test runner in Chrome.

```js
  const line = document.createElement("p");

export function assert(condition, message) {
  if (!condition) {
    throw new Error(message);
  }
}

export function assertEqual(actual, expected, message) {
  if (actual !== expected) {
    throw new Error(`${message} (expected ${expected}, received ${actual})`);
  }
}

export function test(name, fn) {
  try {
    fn();
    appendResult(`PASS ${name}`, "pass");
  } catch (error) {
    appendResult(`FAIL ${name}`, "fail", error.message);
    throw error;
  }
}
```

```js
export const BOARD_SIZE = 19;
const DIRECTIONS = [
  [0, 1],
  [1, 0],
  [1, 1],
  [1, -1],
];

export function createEmptyBoard() {
  return Array.from({ length: BOARD_SIZE }, () => Array(BOARD_SIZE).fill(null));
}

export function createGameState() {
  return {
    board: createEmptyBoard(),
    currentPlayer: "black",
    moveHistory: [],
    winner: null,
    lastMove: null,
  };
}

function isInsideBoard(row, col) {
  return row >= 0 && row < BOARD_SIZE && col >= 0 && col < BOARD_SIZE;
}

function countDirection(board, row, col, rowStep, colStep, stone) {
  let count = 0;
  let nextRow = row + rowStep;
  let nextCol = col + colStep;

  while (isInsideBoard(nextRow, nextCol) && board[nextRow][nextCol] === stone) {
    count += 1;
    nextRow += rowStep;
    nextCol += colStep;
  }

  return count;
}

function findWinner(board, row, col, stone) {
  return DIRECTIONS.some(([rowStep, colStep]) => {
    const total =
      1 +
      countDirection(board, row, col, rowStep, colStep, stone) +
      countDirection(board, row, col, -rowStep, -colStep, stone);

    return total >= 5;
  });
}

function cloneBoard(board) {
  return board.map((row) => [...row]);
}

export function placeStone(state, row, col) {
  if (!isInsideBoard(row, col) || state.winner || state.board[row][col]) {
    return state;
  }

  const board = cloneBoard(state.board);
  const stone = state.currentPlayer;
  board[row][col] = stone;

  const move = { row, col, player: stone };
  const winner = findWinner(board, row, col, stone) ? stone : null;

  return {
    board,
    currentPlayer: winner ? stone : stone === "black" ? "white" : "black",
    moveHistory: [...state.moveHistory, move],
    winner,
    lastMove: move,
  };
}

export function undoMove(state) {
  if (state.moveHistory.length === 0) {
    return state;
  }

  const moveHistory = state.moveHistory.slice(0, -1);
  let nextState = createGameState();

  for (const move of moveHistory) {
    nextState = placeStone(nextState, move.row, move.col);
  }

  return nextState;
}

export function restartGame() {
  return createGameState();
}
```

```js
import { BOARD_SIZE } from "./game.js";

export const BOARD_PADDING = 40;
export const CELL_SIZE = 38.8888888889;
export const STONE_RADIUS = 15;

function drawBoardSurface(ctx, canvas) {
  ctx.clearRect(0, 0, canvas.width, canvas.height);
  ctx.fillStyle = "#c89661";
  ctx.fillRect(0, 0, canvas.width, canvas.height);
}

function drawGrid(ctx) {
  ctx.strokeStyle = "rgba(58, 38, 20, 0.75)";
  ctx.lineWidth = 1;

  for (let index = 0; index < BOARD_SIZE; index += 1) {
    const offset = BOARD_PADDING + index * CELL_SIZE;

    ctx.beginPath();
    ctx.moveTo(BOARD_PADDING, offset);
    ctx.lineTo(BOARD_PADDING + CELL_SIZE * (BOARD_SIZE - 1), offset);
    ctx.stroke();

    ctx.beginPath();
    ctx.moveTo(offset, BOARD_PADDING);
    ctx.lineTo(offset, BOARD_PADDING + CELL_SIZE * (BOARD_SIZE - 1));
    ctx.stroke();
  }
}

function drawStarPoints(ctx) {
  const starPointIndexes = [3, 9, 15];

  ctx.fillStyle = "rgba(50, 32, 19, 0.9)";

  for (const row of starPointIndexes) {
    for (const col of starPointIndexes) {
      ctx.beginPath();
      ctx.arc(
        BOARD_PADDING + col * CELL_SIZE,
        BOARD_PADDING + row * CELL_SIZE,
        4,
        0,
        Math.PI * 2
      );
      ctx.fill();
    }
  }
}

function drawStone(ctx, row, col, stone) {
  const x = BOARD_PADDING + col * CELL_SIZE;
  const y = BOARD_PADDING + row * CELL_SIZE;
  const gradient = ctx.createRadialGradient(x - 4, y - 4, 2, x, y, STONE_RADIUS);

  if (stone === "black") {
    gradient.addColorStop(0, "#5a5a5a");
    gradient.addColorStop(1, "#111111");
  } else {
    gradient.addColorStop(0, "#ffffff");
    gradient.addColorStop(1, "#d8d8d8");
  }

  ctx.beginPath();
  ctx.fillStyle = gradient;
  ctx.arc(x, y, STONE_RADIUS, 0, Math.PI * 2);
  ctx.fill();
}

function drawLastMoveMarker(ctx, lastMove) {
  if (!lastMove) {
    return;
  }

  const x = BOARD_PADDING + lastMove.col * CELL_SIZE;
  const y = BOARD_PADDING + lastMove.row * CELL_SIZE;

  ctx.beginPath();
  ctx.strokeStyle = lastMove.player === "black" ? "#f8fafc" : "#1f2937";
  ctx.lineWidth = 2;
  ctx.arc(x, y, 5, 0, Math.PI * 2);
  ctx.stroke();
}

export function drawGame(canvas, state) {
  const ctx = canvas.getContext("2d");

  drawBoardSurface(ctx, canvas);
  drawGrid(ctx);
  drawStarPoints(ctx);

  for (let row = 0; row < BOARD_SIZE; row += 1) {
    for (let col = 0; col < BOARD_SIZE; col += 1) {
      const stone = state.board[row][col];

      if (stone) {
        drawStone(ctx, row, col, stone);
      }
    }
  }

  drawLastMoveMarker(ctx, state.lastMove);
}
```

```js
import { BOARD_PADDING, CELL_SIZE } from "./render.js";

export function getBoardPositionFromClick(event, canvas) {
  const rect = canvas.getBoundingClientRect();
  const x = event.clientX - rect.left;
  const y = event.clientY - rect.top;
  const col = Math.round((x - BOARD_PADDING) / CELL_SIZE);
  const row = Math.round((y - BOARD_PADDING) / CELL_SIZE);

  if (row < 0 || row >= BOARD_SIZE || col < 0 || col >= BOARD_SIZE) {
    return null;
  }

  const centerX = BOARD_PADDING + col * CELL_SIZE;
  const centerY = BOARD_PADDING + row * CELL_SIZE;
  const distance = Math.hypot(x - centerX, y - centerY);

  return distance <= CELL_SIZE * 0.45 ? { row, col } : null;
}
```

```js
import {
  createGameState,
  placeStone,
  restartGame,
  undoMove,
} from "./game.js";
import { drawGame, BOARD_PADDING, CELL_SIZE } from "./render.js";

const canvas = document.querySelector("#board");
const statusNode = document.querySelector("#status");
const undoButton = document.querySelector("#undo-button");
const restartButton = document.querySelector("#restart-button");

let state = createGameState();

function getBoardPositionFromClick(event, canvasNode) {
  const rect = canvasNode.getBoundingClientRect();
  const x = event.clientX - rect.left;
  const y = event.clientY - rect.top;
  const col = Math.round((x - BOARD_PADDING) / CELL_SIZE);
  const row = Math.round((y - BOARD_PADDING) / CELL_SIZE);

  if (row < 0 || row >= 19 || col < 0 || col >= 19) {
    return null;
  }

  const centerX = BOARD_PADDING + col * CELL_SIZE;
  const centerY = BOARD_PADDING + row * CELL_SIZE;
  const distance = Math.hypot(x - centerX, y - centerY);

  return distance <= CELL_SIZE * 0.45 ? { row, col } : null;
}

function render() {
  drawGame(canvas, state);
  undoButton.disabled = state.moveHistory.length === 0;

  if (state.winner) {
    statusNode.textContent =
      state.winner === "black" ? "Black wins" : "White wins";
    return;
  }

  statusNode.textContent =
    state.currentPlayer === "black" ? "Black to move" : "White to move";
}

canvas.addEventListener("click", (event) => {
  const point = getBoardPositionFromClick(event, canvas);

  if (!point) {
    return;
  }

  const nextState = placeStone(state, point.row, point.col);

  if (nextState !== state) {
    state = nextState;
    render();
  }
});

undoButton.addEventListener("click", () => {
  const nextState = undoMove(state);

  if (nextState !== state) {
    state = nextState;
    render();
  }
});

restartButton.addEventListener("click", () => {
  state = restartGame();
  render();
});

render();
```
